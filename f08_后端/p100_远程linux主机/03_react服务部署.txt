# 下载nodejs
wget https://nodejs.org/dist/v20.11.1/node-v20.11.1-linux-x64.tar.xz
# 解压
tar -xf node-v20.11.1-linux-x64.tar.xz
# 移动到 /usr/local 目录
sudo mv node-v20.11.1-linux-x64 /usr/local/nodejs
# 添加到环境变量
echo "export PATH=/usr/local/nodejs/bin:$PATH" >> ~/.bashrc
source ~/.bashrc


# 检查 Node.js 版本
node --version
# 检查 npm 版本
npm --version

# 安装pnpm
npm install -g pnpm
# 查看pnpm版本
pnpm --version


# 初始化项目 
pnpm install

# 开启项目
pnpm dev

# 查看端口号占用
lsof -i:5173
